\usepackage{glossaries}\newglossaryentry{Frontend}{
	name={Frontend},
	description={Es la parte de la aplicación que se ejecuta en el navegador del usuario}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%               Términos               %%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newglossaryentry{Backend}{
	name={Backend},
	description={Es la parte de la aplicación que se ejecuta en el servidor}
}

\newglossaryentry{BDD}{
	name={BDD},
	description={Del inglés Behavior Driven Development, es una metodología de desarrollo de software que se basa en
	la creación de pruebas que describen el comportamiento de la aplicación}
}

\newglossaryentry{Bot}{
	name={Bot},
	description={Es un programa que se ejecuta en un servidor y que puede interactuar con los usuarios}
}

\newglossaryentry{Bundler}{
	name={Bundler},
	description={Es una herramienta que permite la creación de aplicaciones web a partir de ficheros de código fuente}
}

\newglossaryentry{Cluster}{
	name={Cluster},
	description={Es un conjunto de servidores que trabajan juntos para realizar una tarea}
}

\newglossaryentry{CORS}{
	name={CORS},
	description={Del inglés Cross-Origin Resource Sharing, es un mecanismo que permite que un recurso restringido en
	una página web sea solicitado desde otro dominio externo al que pertenece}
}

\newglossaryentry{Framework}{
	name={Framework},
	description={Es un conjunto de herramientas que facilitan el desarrollo de aplicaciones}
}

\newglossaryentry{Instanciar}{
	name={Instanciar},
	description={Es la acción de crear un objeto en memoria principal}
}

\newglossaryentry{Script}{
	name={Script},
	description={Es un programa que se ejecuta en un intérprete}
}

\newglossaryentry{Docker}{
	name={Docker},
	description={Es una herramienta que permite crear contenedores que ejecutan servicios}
}

\newglossaryentry{Docker Compose}{
	name={Docker Compose},
	description={Es una herramienta que permite definir y ejecutar contenedores Docker}
}

\newglossaryentry{Dockerfile}{
	name={Dockerfile},
	description={Es un fichero que contiene las instrucciones para crear una imagen Docker}
}

\newglossaryentry{Popup}{
	name={Popup},
	description={Es una ventana emergente que se abre en el navegador}
}
